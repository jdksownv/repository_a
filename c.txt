#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(void)
{
    srand(time(NULL));

    int arr[10];

    int i, max = 0;

    printf("일차원 배열 : ");
    for (int i = 0; i < 10; i++)
    {
        arr[i] = rand() % 100 + 1;
        printf("%d ", arr[i]);
        if (max < arr[i])
            max = arr[i];
    }
    printf("\n");
    printf("최대값 : %d", max);
    
}
